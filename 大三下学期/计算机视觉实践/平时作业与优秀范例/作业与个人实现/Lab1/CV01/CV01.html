'''鼠标拾取颜色'''
import numpy as np

img = cv2.imread('Images/Child1.jpg')
colorList = [] #用于存储鼠标拾取的所有像素点的像素值

#事件类型，鼠标位置x，y，鼠标左、中、右键以及Shift、Ctrl、Alt键是否被按下指示器，其他参数
def pointColor(event,x,y,flags,params): 
    if event == cv2.EVENT_LBUTTONDOWN:
        color = img[y,x] #注意坐标系不同
        print(color)
        colorList.append(color)
        
while True:
    cv2.imshow('Image',img)
    cv2.setMouseCallback('Image',pointColor) #窗口名字，回调函数
    
    if cv2.waitKey(1) == ord('q'):
        colorArray = np.array(colorList).reshape((-1,3))
        print("*************************")
        print(colorArray.min(axis = 0))
        print(colorArray.max(axis = 0))
        break
        
cv2.destroyAllWindows()

6.滑动条的创建和使用¶

API：

  • cv2.createTrackbar(trackbarname, winname, value, count, onChange)：创建一个滑动条，参数分别为：滑动条名称，所在窗口名称，初始值，最大值（最小值都是0），回调函数

  • cv2.getTrackbarPos(trackbarname, winname)：获得滑动条的当前值，参数分别为：滑动条名称，所在窗口名称

https://docs.opencv.org/4.x/d7/dfc/group__highgui.html

In [37]:
'''创建滑动条，通过滑动条动态调整颜色范围'''
import cv2
import numpy as np

img=cv2.imread("Images/Child1.jpg")

def empty(a):
    pass

cv2.namedWindow("BGR")
cv2.resizeWindow("BGR", 640, 240)
cv2.createTrackbar("B Min", "BGR", 0, 255, empty) #滑动条名称，所在窗口名称，初始值，最大值，回调函数
cv2.createTrackbar("B Max", "BGR", 255, 255, empty)
cv2.createTrackbar("G Min", "BGR", 0, 255, empty)
cv2.createTrackbar("G Max", "BGR", 255, 255, empty)
cv2.createTrackbar("R Min", "BGR", 0, 255, empty)
cv2.createTrackbar("R Max", "BGR", 255, 255, empty)

while True:
    b_min = cv2.getTrackbarPos("B Min", "BGR") #滑动条名称，窗口名称
    b_max = cv2.getTrackbarPos("B Max", "BGR")
    g_min = cv2.getTrackbarPos("G Min", "BGR")
    g_max = cv2.getTrackbarPos("G Max", "BGR")
    r_min = cv2.getTrackbarPos("R Min", "BGR")
    r_max = cv2.getTrackbarPos("R Max", "BGR")

    lower = np.array([b_min, g_min, r_min])
    upper = np.array([b_max, g_max, r_max])
    mask = cv2.inRange(img, lower, upper)
    cv2.imshow('Mask',mask)
    
    if cv2.waitKey(1) == ord('q'):
        break

cv2.destroyAllWindows()

8.图像位运算¶

8.1 掩码（mask）¶

单通道图像，同时是二值图像（像素值要么为0，要么为255），主要起屏蔽作用，让我们只关注感兴趣区域，即仅让掩膜中的白色区域参与计算，屏蔽黑色部分。

8.2 位运算¶

与、或、异或、非

对应的API分别为：

  • cv2.bitwise_and(src1, src2[, dst[, mask]])
  • cv2.bitwise_or(src1, src2[, dst[, mask]])
  • cv.bitwise_xor(src1, src2[, dst[, mask]])
  • cv2.bitwise_not(src[, dst[, mask]])

注意，src1和src2的shape要一样才能进行计算，除非其中一个为标量。

此外，调用这些函数时，如果设置了mask，则只在mask不为0的区域进行位运算；否则所有区域进行运算。

'''把摄像头获取的视频流保存为一个文件的代码示例'''
import cv2
cap = cv2.VideoCapture(0)

# Define the codec and create VideoWriter object
fourcc = cv2.VideoWriter.fourcc(*'DIVX')  #https://fourcc.org/codecs.php
out = cv2.VideoWriter('Output.avi', fourcc, 20.0, (640,  480)) #filename, fourcc, fps, frameSize

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        print("Can't receive frame (stream end?). Exiting ...")
        break
    
    # write the frame
    out.write(frame)
    cv2.imshow('frame', frame)
    if cv2.waitKey(1) == ord('q'):
        break
# Release everything if job is finished
cap.release()
out.release()
cv2.destroyAllWindows()
